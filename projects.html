Python desktop application using CustomTkinter with automated file sorting, voice commands, secure login, and progress tracking. 

import os
import shutil

def organize_files(path):
    for file in os.listdir(path):
        ext = file.split(".")[-1]
        folder = os.path.join(path, ext)
        os.makedirs(folder, exist_ok=True)
        shutil.move(os.path.join(path, file), folder)
